\section{Preliminares}

\subsection{Teoría Elemental de Conjuntos}

\begin{frame}
\titulos{Noción Intuitiva de Conjunto}{Teoría de Conjuntos}
Para nuestros propósitos de
aplicación bastará con dar la siguiente definición:
\bc
\emph{Un conjunto es una colección bien determinada de objetos.}
\ec\pause
En esta definición con ``bien determinado'' queremos decir que siempre es
posible decidir de manera clara y concisa cuando un objeto dado pertenece o
no a la colección que estamos definiendo.  
\end{frame}

\begin{frame}
  \titulos{Definición de Conjuntos}{Teoría de Conjuntos}
Existen dos maneras de definir conjuntos:
\bi
\item Por \emph{extensión}: dando una lista de todos los elementos del conjunto.\\\pause Por
  ejemplo el conjunto de vocales se define por extensión como
$A=\{a,e,i,o,u\}$.\pause
\item Por \emph{comprensión o intensión}: dando una propiedad que se cumpla
  exactamente para los elementos del conjunto. \\ Por ejemplo, el conjunto de
  vocales se define por comprensión como $A=\{x\;|\;x\;\mbox{es vocal}\;\}$
\ei
\end{frame}


\begin{frame}
  \titulos{El Universo y el Vacío}{Teoría de Conjuntos}
\bi
\item Conjunto Universo: conjunto distinguido el cual nos servirá para
  determinar de manera única el tipo de objetos con los que estamos trabajando. \pause
 Al momento de elegir un universo, denotado $\U$, éste queda fijo y estará
prohibido usar objetos que no le pertenezcan.\espc
\pause
\item Conjunto Vacío: conjunto cuya
característica principal es que no contiene elementos. Se denota
con $\varnothing$.
\ei
\end{frame}


\begin{frame}
 \titulos{La Relación de Pertenencia $\in$ }{Teoría de Conjuntos}

La relación básica para manejar conjuntos es la relación de pertenencia
\beqs
x\in A
\eeqs \pause
ésta es una relación entre un objeto $x$, el cual podría a su
vez ser un conjunto, y un conjunto $A$. \pause
Con respecto a los conjuntos universal y vacío se cumple lo siguiente:
\bi
\item $\fa x(x\in\mathcal{U})$.\espc
\item $\fa x(x\notin\varnothing)$.
\ei

\end{frame}

\begin{frame}
  
\titulos{La Relación de Inclusión $\inc$}{Teoría de Conjuntos}

La segunda relación básica para manejar conjuntos es la \emph{inclusión o
contención} de conjuntos, la cual es una relación definible como sigue:
\beqs
A\inc B\;\Iff \fa x(x\in A\imp x\in B)
\eeqs\pause
Es decir, decimos que $A$ está contenido en $B$ o que $A$ es \emph{subconjunto} de
$B$, lo cual denotamos con $A\inc B$ si todo elemento de $A$ es también
elemento de $B$.\\ \pause
Obsérvese que si $A=B$ entonces en particular $A\inc B$. Si $A\inc B$ y $A\neq
B$ entonces decimos que $A$ es un subconjunto propio de $B$ y escribimos $A\subset B$ 
\end{frame}

\begin{frame}
\titulos{Propiedades de la Inclusión}{Teoría de Conjuntos}
\bi
\item $A\inc\U$
\item $\varnothing\inc A$\espc
\item Extensionalidad $A=B$ si y sólo si $A\inc B$ y $B\inc A$.
\item Reflexividad: $A\inc A$\espc
\item Transitividad: Si $A\inc B$ y $B\inc C$ entonces $A\inc C$.
\item Subconjunto Propio: $A\subset B$ si y sólo si $A\inc B$ y $A\neq B$.
\ei
\end{frame}

%\subsubsection{Operaciones Básicas}

\begin{frame}


\titulos{La Unión}{Operaciones Básicas}
La unión de dos conjuntos $A$ y $B$ es el conjunto que cuyos elementos están
en $A$ ó en $B$.\pause
\beqs
A\cup B=\{x\;|\;x\in A\;\mbox{ó}\;x\in B\}
\eeqs \pause
%\subsubsection{Propiedades de la Unión}
\bi
\item Idempotencia: $A\cup A=A$.
\item Simetria o Conmutatividad: $A\cup B=B\cup A$.
\item Asociatividad: $A\cup (B\cup C)=(A\cup B)\cup C$.
\item Neutro: $A\cup\U=\U$
\item Identidad: $A\cup\vacio=A$.
\item $A\inc A\cup B,\;B\inc A\cup B$.
\item $A\inc B$ si y sólo si $A\cup B=B$.
\ei
\end{frame}
\begin{frame}
\titulos{La Intersección}{Operaciones Básicas}
La intersección de dos conjuntos $A$ y $B$ es el conjunto que cuyos elementos estaban
en $A$ {\bf y} en $B$.\pause
\beqs
A\cap B=\{x\;|\;x\in A\;\mbox{y}\;x\in B\}
\eeqs
\pause
\bi
\item Idempotencia: $A\cap A=A$.
\item Simetria o Conmutatividad: $A\cap B=B\cap A$.
\item Asociatividad: $A\cap (B\cap C)=(A\cap B)\cap C$.
\item Neutro: $A\cap\vacio=\vacio$
\item Identidad: $A\cap\U=A$.
\item $A\cap B\inc A,\;A\cap B\inc B$.
\item $A\inc B$ si y sólo si $A\cap B=A$.
\ei
\end{frame}
\begin{frame}
\titulos{Distributividad}{entre unión e intersección}

La unión y la intersección se relacionan mediante las propiedades de
distributividad:\pause
\bi
\item $A\cup(B\cap C)=(A\cup B)\cap (A\cup C)$.\espc
\item $A\cap(B\cup C)=(A\cap B)\cup (A\cap C)$.
\ei
\end{frame}

\begin{frame}
\titulos{La Potencia}{Operaciones Básicas}
La potencia de un conjunto es el conjunto de todos los subconjuntos de $A$.
\beqs
\Pe(A)=\{X\;|\;X\inc A\}
\eeqs\pause
\bi
\item $\Pe(A)\neq\vacio$.
\item $\vacio\in\Pe(A)$.
\item $A\in\Pe(A)$.
\item Si $A\inc B$ entonces $\Pe(A)\inc\Pe(B)$.
\ei
\end{frame}

%\subsubsection{Otras Operaciones}

\begin{frame}
\titulos{El Complemento}{Otras Operaciones}
El complemento de un conjunto $A$ es el conjunto de elementos del universo
que {\bf no} pertenecen a $A$.
\beqs
\ol{A}=\{x\in\U\;|\;x\notin A\}
\eeqs\pause
\bi
\item Idempotencia: $\ol{\ol{A}}=A$.
\item De Morgan: $\ol{A\cup B}=\ol{A}\cap\ol{B}$.
\item De Morgan: $\ol{A\cap B}=\ol{A}\cup\ol{B}$.
\item $A\cup\ol{A}=\U$
\item $A\cap\ol{A}=\vacio$
\item $\ol{\U}=\vacio$
\item $\ol{\vacio}=\U$.
\item Si $A\inc B$ entonces $\ol{B}\inc\ol{A}$.
\ei
\end{frame}
\begin{frame}
\titulos{La Diferencia}{Otras Operaciones}
La diferencia de dos conjuntos $A$ y $B$, denotada $A-B$, es el conjunto de
todos los elementos de $A$ que {\bf no} están en $B$.
\beqs
A-B=\{x\;|\;x\in A\;\mbox{y}\;x\notin B\}
\eeqs

\bi
\item $A-B\inc A$
\item $A-A=\vacio$
\item $A-\vacio=A$
\item $A-\U=\vacio$
\item $\U-A=\ol{A}$
\item $(A-B)\cap B=\vacio$
\item $(A-B)\cup B=A\cup B$
\item $A-B=A\cap\ol{B}$
%\item $A-(B\cup C)=(A-B)\cap(A-C)$
%\item $A-(B\cap C)=(A-B)\cup(A-C)$
\ei
\end{frame}
\begin{frame}
\titulos{El Producto Cartesiano}{Otras Operaciones}
El producto cartesiano de dos conjuntos $A,B$ se define como
\beqs
A\times B=\{\pt{x,y}\;|\;x\in A\;\mbox{y}\;y\in B\}
\eeqs\pause
Es decir, $A\times B$ es el conjunto de
todas las parejas ordenadas cuyo primer elemento pertenece a $A$ y el segundo
elemento pertenece a $B$.\\\pause
\beqs
\pt{x,y}\in A\times B\;\mbox{si y sólo si}\;x\in A\;\mbox{y}\;y\in B
\eeqs
\pause
Obsérvese que debido a que las parejas tienen un orden, si $A\neq B$ entonces
$A\times B\neq B\times A$.\\ \pause
En general se puede definir el producto cartesiano de $n$ conjuntos, denotado
$A_1\times A_2\times\ldots\times A_n$ análogamente.
\end{frame}
\begin{frame}
\titulos{Propiedades del Producto Cartesiano}{Otras Operaciones}
\bi
\item Si $A=\varnothing$ entonces $A\times B=B\times A=\varnothing$.
\item Si $A\times B=B\times A$ entonces $A=\varnothing$ ó $B=\varnothing$ ó
  $A=B$.
\item $A\times (B\cup C)=(A\times B)\cup (A\times C)$
\item $(A\cup B)\times C=(A\times C)\cup (B\times C)$
\item $A\times (B\cap C)=(A\times B)\cap (A\times C)$
\item $(A\cap B)\times C=(A\times C)\cap (B\times C)$
\item $A\times (B- C)=(A\times B)- (A\times C)$
\ei
\end{frame}


\subsection{Relaciones Binarias}
\begin{frame}\titulos{Relaciones Binarias}{Definición}
\bi
\item  Una relación binaria $R$ de $A$ a $B$ es un conjunto de parejas ordenadas
donde el primer elemento pertenece a $A$ y el segundo elemento pertenece a
$B$.

\item Formalmente una relación binaria $R$ de $A$ a $B$ es un subconjunto del producto
cartesiano $A\times B$, es decir $R\inc A\times B$.
\item Si $\pt{a,b}\in R$
escribimos a veces también $a\;R\;b$.
\item Si $A=B$ entonces $R\inc A\times A$ y decimos simplemente que
  $R$ es una relación binaria sobre $A$.
  \ei
\end{frame}

\begin{frame}
\titulos{Operaciones con Relaciones}{}
Como las relaciones son también conjuntos entonces podemos aplicarles todas
las operaciones ya conocidas de conjuntos. En particular son mas usadas la
unión, intersección, complemento y diferencia de relaciones.\\
Recordemos las definiciones en el caso particular de relaciones, si $R,S\inc
A\times B$ son dos relaciones de $A$ a $B$ entonces 
\bi
\item $R\cup S=\{\pt{x,y}\;|\;\pt{x,y}\in R\;\mbox{ó}\;\pt{x,y}\in S\}$
\item $R\cap S=\{\pt{x,y}\;|\;\pt{x,y}\in R\;\mbox{y}\;\pt{x,y}\in S\}$
\item $\ol{R}=\{\pt{x,y}\;|\;\pt{x,y}\notin R\}$
\item $R-S=\{\pt{x,y}\;|\;\pt{x,y}\in R\;\mbox{y}\;\pt{x,y}\notin S\}$
\ei
\end{frame}
\begin{frame}\titulos{Operaciones con Relaciones}{Inversa y Composición}
Adicionalmente tenemos dos operaciones para relaciones que NO se definen para
conjuntos en general, la relación inversa y la composición de relaciones, la
cual sólo se define bajo condiciones especiales.\pause

\bi
\item Si $R\inc A\times B$ entonces la inversa de $R$, denotada $R^{-1}$ es
  una relación de $B$ a $A$, es decir, $R^{-1}\inc B\times A$, definida como sigue:
\beqs
R^{-1}=\{\pt{y,x}\;|\;\pt{x,y} \in R\}
\eeqs
\ei
\end{frame}
\begin{frame}
\titulos{Operaciones con Relaciones}{Inversa y Composición}
\bi
\item Si $R\inc A\times B$ y $S\inc B\times C$ entonces la composición de $R$
  y $S$ denotada $R\circ S$ es una relación de $A$ a $C$, es decir, $R\circ
  S\inc A\times C$ definida como sigue:\pause
\beqs
R\circ S=\{\pt{x,z}\;|\;\pt{x,y}\in R\;\mbox{y}\;\pt{y,z}\in S,\mbox{para
  algún}\;y\in B\}
\eeqs
\ei
\end{frame}
%\subsubsection{Tipos de Relaciones}

\begin{frame}
\titulos{Tipos de Relaciones}{Relaciones Binarias}

Si $R$ es una relación \alert{binaria} sobre un conjunto $A$ podemos hablar de las
siguientes propiedades de $R$:

\bi
\item $R$ es reflexiva si $\fa x\in A\;(\pt{x,x}\in R)$\espc
\item $R$ es simétrica si $\fa x,y\in A\;(\pt{x,y}\in R\imp\pt{y,x}\in R)$\espc
\item $R$ es transitiva si $\fa x,y,z\in A\;(\pt{x,y}\in R\land\pt{y,z}\in
  R\imp\pt{x,z}\in R)$\espc
\item $R$ es relación de equivalencia si $R$ es reflexiva, simétrica y transitiva.
  \ei
  \end{frame}
  \begin{frame}
\titulos{Tipos de Relaciones}{Relaciones Binarias}
  \bi
\item $R$ es antirreflexiva si $\fa x\in A\;(\pt{x,x}\notin R)$\espc
\item $R$ es antisimétrica si $\fa x,y\in A\;(\pt{x,y}\in
  A\;\land\pt{y,x}\in A\imp x=y)$\espc
\item $R$ es asimétrica si $\fa x,y\in A\;(\pt{x,y}\in R\imp
  \pt{y,x}\notin R)$
\ei
\end{frame}


\subsection{Funciones}

\begin{frame}
\titulos{Funciones}{Definición}  

Una función $f$ de $A$ en $B$, denotada $f:A\imp B$ es una relación $f\inc
A\times B$ tal que para cualquier $x\in A$ existe {\bf un único} $y\in B$ tal que 
$\pt{x,y}\in f$. 
En tal caso decimos que $y$ es la imagen de $x$ bajo $f$ y escribimos
$y=f(x)$.\\
%Obsérvese que $f$ es función si y sólo si se cumple que
%\beqs
%\fa x\fa x\;(x=z\imp f(x)=f(z))
%\eeqs
\pause
Dada una función $f:A\imp B$ definimos los siguientes conjuntos:
\bi
\item El dominio de $f$, denotado $dom(f)$ es:
\beqs
dom(f)=A
\eeqs
El dominio de $f$ es el conjunto de primeras componentes de los pares
de $f$.
\ei
\end{frame}

\begin{frame}
\titulos{Funciones}{Definición}
\bi
\item El codominio, contradominio o rango de $f$, denotado $codom(f)$ o
  $ran(f)$ es el conjunto $B$.
\beqs
codom(f)=B
\eeqs
\item La imagen de $f$, denotada $Im(f)$  es el subconjunto de $B$ que consta de las segundas
  componentes de $f$:
\beqs\ba{rl}
Im(f)= & \{y\in B\;|\;\ex x\in A\;(\pt{x,y}\in f)\} \\
     = &\{y\in B\;|\;\ex x\in A\;(f(x)=y))\}
     \ea
\eeqs
\ei
\end{frame}
%\subsubsection{Tipos de Funciones}
\begin{frame}\titulos{Tipos de Funciones}{}
De acuerdo a ciertas propiedades del dominio o codominio una función $f:A\imp
B$ se puede clasificar como sigue:
\bi
\item $f$ es una función \emph{parcial} si algún elemento de $dom(f)=A$ no tiene
  imagen en $B$. \\ Por ejemplo la función $f:\R\to\R$ definida por
  \beqs f(x)=\frac{1}{x}\eeqs
es una función parcial pues $f(0)$ no está definida.
\ei
\end{frame}
\begin{frame}\titulos{Tipos de Funciones}{}
  \bi
\item $f$ es una función \emph{inyectiva} o \emph{``uno a uno''} si elementos
  distintos del dominio de $f$ van a dar a imagenes distintas, o
  equivalentemente si dos elementos del dominio tienen la misma imagen
  entonces es que son el mismo elemento. Formalmente, $f$ es inyectiva si:
\beqs
\fa x\fa z\;(f(x)=f(z)\imp x=z)
\eeqs
\ei
\end{frame}
\begin{frame}\titulos{Tipos de Funciones}{}
  \bi
\item $f$ es \emph{suprayectiva} o \emph{sobre} si $codom(f)=Im(f)$ es
  decir,si cualquier elemento del codominio es imagen bajo $f$ de algun
  elemento del dominio. Formalmente, $f$ es sobre si:
\beqs
\fa y\in B\;\ex x\in A\;(f(x)=y)
\eeqs
\item $f$ es \emph{biyectiva} si $f$ es a la vez inyectiva y
  suprayectiva. 
\ei
\end{frame}
