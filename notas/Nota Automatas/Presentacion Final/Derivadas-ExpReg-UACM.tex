\begin{frame}
\titulos{Alfabeto}{Preliminares}
\begin{definition}
Un \textit{alfabeto} $\Sigma$ es un conjunto finito y ordenado de símbolos a partir del cual se construyen palabras
\end{definition}
\begin{example}
\[
\Sigma = \{a ,b, c, d\} \qquad \text{palabra } \quad aabbbccccddddd
\]
\end{example}
\begin{example}
\[
\Sigma = \{ 0, 1 \} \qquad \text{palabra } \quad 10101010101
\]
\end{example}
\begin{example}
\[
\Sigma = \{ a, e, i, o, u \} \qquad \text{palabra } \quad auiaue
\]
\end{example}
\end{frame}

\begin{frame}
\titulos{Lenguaje formal}{Preliminares}
\begin{definition}
Un \textit{lenguaje formal}, es el conjunto de palabras $u,v,w$ de longitud finita formadas a partir de un alfabeto finito $\Sigma$
\end{definition}
\begin{example}
El lenguaje de todas la palabras sobre el alfabeto $\Sigma = \{0, 1 \}$ tales que las palabras sólo constan de ceros y un uno al final.
\[
\{ 01, 001, 0001, 00001, 000001, 0000001, \dots \}
\]
\end{example}
\begin{example}
El conjunto de palabras sobre el alfabeto $\Sigma = \{ a \}$ tales que, las palabras tienen longitud par
\[
\{ \varepsilon, aa, aaaa, aaaaaa, aaaaaaaa, aaaaaaaaaa, \dots \}
\]
\end{example}
\end{frame}

\begin{frame}
\titulos{Autómata}{Preliminares}
\begin{definition}
Un autómata finito es un modelo matemático de una máquina que acepta cadenas de un lenguaje definido sobre un alfabeto $\Sigma$.

\vspace*{3mm}
Formalmente un autómata finito se define como una 5--tupla de la forma
\[
FA = (Q, \Sigma, \delta, q_0, F)
\]
donde\\
\begin{itemize}
\item $Q$: conjunto finito de estados
\item $\Sigma$: alfabeto o conjunto finito de símbolos de entrada
\item $\delta$: función de transición de estados, que se define como:
\[
\delta \colon Q \times \Sigma \to Q
\]
\item $q_0$: estado inicial; $q_0 \in Q$
\item $F$: conjunto de estados finales o estados de aceptación; $F \subseteq Q$
\end{itemize}
\end{definition}
\end{frame}

\begin{frame}
\titulos{Autómata}{Preliminares}
\begin{example}
Este autómata reconoce cadenas de $a$'s y $b$'s, tales que el número de $b$'s es múltiplo de tres.\\
Donde: $FA = (Q,\Sigma, \delta, q_0, F)$.
\end{example}

\begin{frame}
\titulos{Lenguaje Regular}{Preliminares}
\begin{definition}
Un lenguaje $\mathcal{L}$ es regular si y sólo si, existe un autómata finito que lo acepta.
\end{definition}
\begin{example}
\[
\mathcal{L} = \{a^i b^j \; | \; i \ge 0, \; j \ge 0 \}
\]
Esto es suponiendo que el alfabeto es $\Sigma = \{ a, b \}$
\end{example}

\begin{frame}
\titulos{Expresiones Regulares}{Preliminares}
\begin{definition}
Dado un alfabeto $\Sigma$, las \textit{expresiones regulares} sobre $\Sigma$ están formadas por las siguientes reglas recursivas:
\begin{itemize}
\item $\varnothing$
\item $\varepsilon$
\item $a$, $a \in \Sigma$
\end{itemize}
Si $r$ y $s$ son expresiones regulares,
\begin{itemize}
\item $r + s$
\item $r \cdot s$
\item $r^{\ast}$
\end{itemize}
Éstas y sólo estas son expresiones regulares
\end{definition}
\begin{example}
\[
((a + b)^{\ast} c (a + b)^{\ast} c)^{\ast} (a + b)^{\ast}
\]
\end{example}
\end{frame}

\begin{frame}
\titulos{Expresiones regulares}{Lenguaje asociado}
\begin{definition}
El \textit{lenguaje asociado a una expresión regular} $r$, es el conjunto de cadenas $\mathcal{L} \llbracket r \rrbracket \subseteq \Sigma^{\ast}$ generado por las siguientes reglas
\begin{align*}
\mathcal{L} \llbracket \varnothing \rrbracket &= \varnothing\\
\mathcal{L} \llbracket \varepsilon \rrbracket &= \{ \varepsilon \}\\
\mathcal{L} \llbracket a \rrbracket &= \{ a\}\\
\mathcal{L} \llbracket r \cdot s \rrbracket &= \{ u \cdot v | u \in \mathcal{L} \llbracket r \rrbracket \textrm{ y } v \in L \llbracket s \rrbracket \}\\
\mathcal{L} \llbracket r + s \rrbracket &= \mathcal{L} \llbracket r \rrbracket \cup \mathcal{L} \llbracket s \rrbracket\\
\mathcal{L} \llbracket r^{\ast} \rrbracket &= \{ \varepsilon \} \cup \mathcal{L} \llbracket r \cdot r^{\ast} \rrbracket\\
\end{align*}
\end{definition}
\begin{theorem}
$\mathcal{L}$ es regular si y sólo si existe $r$ expresión regular tal que $\mathcal{L} = \mathcal{L} \llbracket r \rrbracket$
\end{theorem}
\end{frame}

\begin{frame}
\titulos{Derivadas}{Preliminares}
\begin{definition}
La \textit{derivada de un lenguaje} $\mathcal{L} \subseteq \Sigma^{\ast}$ con respecto a una cadena $u \in \Sigma^{\ast}$ se define como
\[
\partial_u \mathcal{L} = \{ v | u \cdot v \in \mathcal{L} \}
\]
\end{definition}
\begin{example}
\[
\mathcal{L} = \{ ab, abb, abbb, abbbb, abbbbb, abbbbbb, \dots \} = \mathcal{L} \llbracket (ab) \cdot b^{\ast} \rrbracket
\]
\[
\partial_{ab} \mathcal{L} = \{ \varepsilon, b, bb, bbb, bbbb, bbbbb, \dots \} = \mathcal{L} \llbracket b^{\ast} \rrbracket
\]
\end{example}
\begin{theorem}
Si $\mathcal{L} \subseteq \Sigma^{\ast}$ es regular, entonces $\partial_u \mathcal{L}$ también es regular.
\end{theorem}
\end{frame}

\begin{frame}
\titulos{Derivadas}{Problema}
\begin{problem}
Dada una expresión regular $r$ y una cadena $u \in \Sigma^{\ast}$, determinar si $u \in L \llbracket r \rrbracket$.
\end{problem}

Resolver el problema es equivalente a resolver:
\[
u \in L \llbracket r \rrbracket \Leftrightarrow \varepsilon \in L \llbracket \partial_u r \rrbracket
\]
Lo cual sucede si y sólo si
\[
\nu(\partial_u r) = \varepsilon
\]
\end{frame}

\begin{frame}
\titulos{Derivadas}{}
Decimos que una expresión regular $r$ es anulable, si el lenguaje definido contiene a la cadena vacía, esto es si $\varepsilon \in \mathcal{L} \llbracket r \rrbracket$ la función $\nu : RE \to RE$ tiene la siguiente propiedad:
\[
\nu(r) = \left\{
\begin{array}{ll}
\varepsilon & \text{Si } r \text{ es anulable}\\
\varnothing & \text{otro caso}
\end{array}
\right.
\]
Y está definida como sigue
\begin{align*}
\nu(\varepsilon) &= \varepsilon\\
\nu(\varnothing) &= \varnothing\\
\nu(a) &= \varnothing\\
\nu(r \cdot s) &= \nu(r) \cap \nu(s)\\
\nu(r + s) &= \nu(r) + \nu(s)\\
\nu(r^{\ast}) &= \varepsilon\\
\end{align*}
\end{frame}

\begin{frame}
\titulos{Derivadas}{Problema}
Usando lo anterior y la definición de $\partial_u$, generamos un algoritmo que verifique si $u \in L \llbracket r \rrbracket$. 
\begin{definition}
Expresandolo en términos de la relación $r \sim u$ ($u$ caza con $r$).
\begin{gather*}
r \sim \varepsilon \Leftrightarrow \nu(r) = \varepsilon\\
r \sim a \cdot \omega \Leftrightarrow \partial_a r \sim \omega
\end{gather*}
\end{definition}
\begin{theorem}
\[
r \sim u \Leftrightarrow u \in L \llbracket r \rrbracket
\]
\end{theorem}
\end{frame}

\begin{frame}
\titulos{Derivadas}{Expresiones regulares}
Reglas de Brzozowski para expresiones regulares con respecto a un símbolo $a \in \Sigma^{\ast}$.
\begin{align*}
\partial_a \varepsilon &= \varnothing\\
\partial_a a &= \varepsilon\\
\partial_a b &= \varnothing \; \text{si } b \ne a\\
\partial_a \varnothing &= \varnothing\\
\partial_a (r \cdot s) &= (\partial_a r) \cdot s + \nu(r) \cdot \partial_a s\\
\partial_a (r + s) &= \partial_a r + \partial_a s\\
\partial_a (r^{\ast}) &= \partial_a r \cdot r^{\ast}
\end{align*}
La reglas se puede extender a cadenas como:
\begin{gather*}
\partial_{\varepsilon} r = r\\
\partial_{u \cdot a} r = \partial_a ( \partial_u r)
\end{gather*}
\end{frame}

\textbf{fun} goto $q \;(c,(Q,\delta))=$\\
$\qquad \textbf{let  } q_c = \partial_c q$\\
$\qquad \textbf{in}$\\
$\qquad \quad \textbf{if } \exists q' \in Q$ tal que $q' \equiv q_c$\\
$\qquad \qquad \textbf{then } (Q, \delta \cup \{(q,c) \to q'\})$\\
$\qquad \quad \textbf{else }$\\
$\qquad \qquad \textbf{let } Q' = Q \cup \{q_c\}$\\
$\qquad \qquad \textbf{let } \delta' = \delta \cup \{(q,c) \to q_c\}$\\
$\qquad \qquad \textbf{in } \text{explora }(Q',\delta',q_c)$\\
$\phantom{a}$\\
\textbf{and } explora $(Q,\delta,q) = \text{fold }(\text{goto } q) \; (Q,\delta) \;\; \Sigma$\\
$\phantom{a}$\\
\textbf{fun} mkDFA $r =$\\
$\qquad \textbf{let  } q_0 = \partial_{\varepsilon}r$\\
$\qquad \textbf{let  } (Q,\delta) = \text{explora } (\{q_0\}, \{ \}, q_0)$\\
$\qquad \textbf{let  } F = \{ q | q \in Q \text{ y } \nu(q) = \varepsilon \}$\\
$\qquad \textbf{in  } (Q,\Sigma, q_0,F,\delta)$\\
